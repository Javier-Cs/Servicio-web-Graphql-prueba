
// obtener todos los productos
query{
  listarProductos{
    id,nombre, precio,categoria {
      id
    }
  }
}




// obtener un producto mediante su id
query{
  listarById(id: 81){
    nombre, categoria{
      nombre}
  }
}



// guardar un producto usando una mutation
mutation{
  guardarProducto(
    productoDto:  {
       nombre: "juaquin",
       precio: 340,
       cantidad: 6,
       categoriaId: 5
    }
  ){
    id,nombre,precio,cantidad,categoria{
      nombre
    }
  }
}
